Актуальная структура данных
api/
├── config.php          Данные для входа в бд
├── login.php           Логин
├── logout.php          Выход
├── register.php        Регистрация
├── create_db.php       Скрипт для пересоздания бд
├── admin/
│   ├── halls_create.php   # Создание зала
│   ├── halls_update.php   # Редактирование зала
│   ├── halls_delete.php   # Удаление зала
│   └── halls_list.php     # Список залов (для админки)
├── user/
│   ├── halls.php           Список залов с фильтрацией
│   ├── hall_detail.php     Детали зала + слоты времени
│   └── booking_create.php  Создание брони пользователем
└── uploads/
    └── halls/
         └── (id зала, кроме тестовых данных)           # фотографии(пока локально)
